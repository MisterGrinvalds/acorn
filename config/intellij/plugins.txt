# IntelliJ IDEA Plugin List
# Reference list for manual installation
#
# Install via: Settings > Plugins > Marketplace
# Or via command line: idea installPlugins <plugin-id>

org.jetbrains.plugins.go           # Go

org.intellij.scala                 # Scala
org.toml.lang                      # TOML
org.jetbrains.plugins.yaml         # YAML (bundled)

# Framework Support
com.jetbrains.plugins.webDeployment # Remote Hosts Access
org.jetbrains.kotlin               # Kotlin (bundled)
com.dmarcotte.handlebars           # Handlebars/Mustache
org.jetbrains.plugins.vue          # Vue.js

com.intellij.database              # Database Tools (bundled in Ultimate)

# Version Control
Git4Idea                           # Git Integration (bundled)
org.jetbrains.plugins.github       # GitHub Integration (bundled)
com.github.copilot                 # GitHub Copilot

# Code Quality
CheckStyle-IDEA                    # CheckStyle integration
org.sonarlint.idea                 # SonarLint
com.intellij.plugins.watcher       # File Watchers

com.github.copilot                # GitHub Copilot

.ignore                           # .gitignore support
